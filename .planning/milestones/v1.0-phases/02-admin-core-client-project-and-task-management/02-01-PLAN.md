  <name>Task 3: Create client list page with project counts and create form</name>
  <files>app/admin/clients/page.tsx, app/admin/clients/actions.ts</files>
  <action>
    Create `app/admin/clients/page.tsx` (RSC server component):

    1. Import `createClient` from `@/lib/supabase/server`.
    2. Query all clients ordered by created_at desc:
       ```sql
       SELECT clients.*, COUNT(projects.id) FILTER (WHERE projects.status = 'active') as active_projects_count
       FROM clients
       LEFT JOIN projects ON projects.client_id = clients.id
       GROUP BY clients.id
       ORDER BY clients.created_at DESC
       ```
    3. Display as a table/card list with: client name, active project count badge, created date.
    4. "Add Client" button that opens a dialog (shadcn Dialog) with a single text input for client name.
    5. Form action calls `createClientAction` from `app/admin/clients/actions.ts`.
    6. Each client row has a delete button (calls `deleteClientAction`).

    Create `app/admin/clients/actions.ts`:
    ```typescript
    'use server'
    import { revalidatePath } from 'next/cache'
    import { redirect } from 'next/navigation'
    import { createClient } from '@/lib/supabase/server'
    import crypto from 'node:crypto'

    export async function createClientAction(formData: FormData) {
      const name = formData.get('name') as string
      if (!name || name.trim().length < 2) return { error: 'Client name must be at least 2 characters' }

      const slug = crypto.randomBytes(8).toString('hex')  // 16-char unique slug
      const supabase = await createClient()
      const { error } = await supabase.from('clients').insert({ name: name.trim(), slug })
      if (error) return { error: error.message }

      revalidatePath('/admin/clients')
      redirect('/admin/clients')
    }

    export async function deleteClientAction(clientId: string) {
      const supabase = await createClient()
      const { error } = await supabase.from('clients').delete().eq('id', clientId)
      if (error) return { error: error.message }
      revalidatePath('/admin/clients')
    }
    ```
  </action>
